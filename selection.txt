# apps/selection/models.py
import json

from django.db import models

from apps.control_manager.utils.utils import (
    CustomJSONEncoder,
)
from shared.base_classes.base_model import (
    BaseModel,
)


class SelectionStatus(models.TextChoices):
    """
    Класс для перечисления возможных состояний подборки.
    С его помощью можно:
    1) отслеживать, на каком этапе находится генерация подборки;
    2) избежать дублирующего создания одной и той же подборки (как ты описала ранее);
    3) упорядочивать логику очередей и ответов пользователям.
    """
    QUEUE = "queue", "Только что поставили задачу, воркер ещё не взял."
    PROCESS = "process", "Подборка создается. Запросы отправляются, PDF генерится."
    ON_REVIEW = "on_review", "PDF готов, но модератор ещё не проверил."
    DONE = "done", "Подборка прошла модерацию, можно отправлять пользователю."
    DEVIATION = "deviation", "Подборка отклонена. Пользователю уходит объяснение, но не PDF."
    FAILED = "failed", "Ошибка создания подборки."


class SelectionTaskType(models.TextChoices):
    """
    Тип задачи, по которой создаётся подборка.
    """
    COMPOSITION_ANALYSIS = (
        "detailed_analysis",
        "Подробный анализ каждого элемента состава")
    BEST_CHOICE = "best", "Лучшее средство"
    ANALOG = "analog", "Аналог"


class Selection(BaseModel):
    """
    Модель Подборки какого-то средства (продукта) Золотого Яблока
    """
    product = models.ForeignKey(
        "products.Product",
        on_delete=models.CASCADE,
        related_name="selections",
        verbose_name="Средство",
    )

    users = models.ManyToManyField(
        "users.User",
        related_name="selections",
        verbose_name="Пользователи, у которых есть эта подборка",
    )

    task_type = models.CharField(
        max_length=50,
        choices=SelectionTaskType.choices,
        verbose_name="Тип задачи подборки",
    )

    selection_status = models.CharField(
        max_length=20,
        choices=SelectionStatus.choices,
        default=SelectionStatus.QUEUE,
        verbose_name="Статус генерации подборки"
    )

    pdf_url = models.URLField(
        "Ссылка на PDF-файл подборки",
        blank=True,
        null=True,
    )

    request_details = models.TextField(
        null=True,
        blank=True,
        verbose_name="Отправленные данные в OpenAI для анализа"
    )

    final_analysis = models.TextField(
        null=True,
        blank=True,
        verbose_name="Финальный ответ от OpenAI"
    )

    price = models.JSONField(
        null=True,
        blank=True,
        verbose_name="Стоимость запроса к OpenAI"
    )

    error_message = models.TextField(
        blank=True,
        null=True,
        verbose_name="Сообщение об ошибке",
    )

    def set_request_details(
            self,
            json_str: str,
    ) -> None:
        """
        Принимает запросы, отправленные в OPENAI в виде JSON-строки
        и сохраняет в атрибут request_details.

        :param json_str: JSON-строка с запросами в OPENAI
        :return: None
        """

        self.request_details = json_str
        self.save(update_fields=["request_details"])

    def set_final_analysis(
            self,
            json_str: str,
    ) -> None:
        """
        Принимает ответы от OPENAI в виде JSON-строки
        и сохраняет в final_analysis.

        :param json_str: JSON-строка с ответами от OPENAI
        :return: None
        """

        self.final_analysis = json_str
        self.save(update_fields=["final_analysis"])

    def finish(
            self,
            pdf_url: str,
    ) -> None:
        """
        Записывает ссылку на PDF в атрибут и переводит статус подборки в DONE.

        :param pdf_url: Строка со ссылкой на PDF
        :return: None
        """

        self.pdf_url = pdf_url
        self.selection_status = SelectionStatus.DONE
        self.save(update_fields=["pdf_url", "selection_status"])

    def calculate_price(
            self,
            price_request: dict,
    ) -> None:
        """
        Записывает в атрибут подборки информацию по стоимость этой подборки
        :param price_request: словарь с ценами и токенами на запрос, ответ, общую стоимость
        :return: None
        """
        self.price = json.loads(json.dumps(price_request, cls=CustomJSONEncoder))
        self.save(update_fields=["price"])

    def mark_as_processing(self) -> None:
        """
        Меняем статус подборки на PROCESS, то есть подборка начала готовиться.
        :return: None
        """
        self.selection_status = SelectionStatus.PROCESS
        self.save(update_fields=["selection_status"])

# apps/selection/selectors/selection.py

from apps.selection.models import (
    Selection,
    SelectionStatus,
    SelectionTaskType,
)


def get_selection_by_product_and_task_type(
        link_ga: str,
        task_type: str | SelectionTaskType,
) -> Selection | None:
    """
    Получение подборки по ссылке на продукт и типу задачи.

    :param link_ga: Ссылка на средство в Золотом Яблоке
    :param task_type: тип задачи
    :return: объект Selection или None, если не найден
    """
    try:
        return Selection.objects.select_related("product").get(
            product__link_ga=link_ga,
            task_type=task_type,
        )
    except Selection.DoesNotExist:
        return None


def get_selection_by_id(
        selection_id: int,
) -> Selection | None:
    """
    Получение подборки по id.
    :param selection_id: Id подборки
    :return:
    """
    try:
        return Selection.objects.select_related("product").get(id=selection_id)
    except Selection.DoesNotExist:
        return None

def get_ready_selection_for_user(
    user,
    selection,
) -> str | None:
    """
    Проверяет, есть ли у пользователя готовая подборка,
    и можно ли её сразу отдать.
    """
    if not selection:
        return None

    if (
        selection.selection_status == SelectionStatus.DONE
        and selection.users.filter(pk=user.pk).exists()
    ):
        return "Эта подборка у Вас уже есть. Отправляем PDF."

    return None

        self.assertNotEqual(selection.selection_status, SelectionStatus.DONE)
        self.assertIsNone(selection.pdf_url)

        pdf_url = "https://example.com/file.pdf"
        selection.finish(pdf_url=pdf_url)

        self.assertEqual(selection.selection_status, SelectionStatus.DONE)
        self.assertEqual(selection.pdf_url, pdf_url)

    def test_calculate_price_saves_data(self):
        """
        Проверка, что метод calculate_price корректно сохраняет данные в поле price.
        """
        selection = Selection.objects.create(
            product=self.product,
            task_type=SelectionTaskType.BEST_CHOICE,
        )

        price_data = {
            "prompt_tokens": 120,
            "completion_tokens": 340,
            "total_tokens": 460,
            "price_usd": 0.0123,
        }

        # вызываем метод
        selection.calculate_price(price_request=price_data)

        # обновим объект из базы
        selection.refresh_from_db()

        # проверим, что данные сохранились
        self.assertEqual(selection.price["prompt_tokens"], 120)
        self.assertEqual(selection.price["completion_tokens"], 340)
        self.assertEqual(selection.price["total_tokens"], 460)
        self.assertAlmostEqual(selection.price["price_usd"], 0.0123)

    def test_mark_as_processing_updates_status(self):
        """
        Проверка, что метод mark_as_processing
        переводит подборку в статус PROCESS и сохраняет это в БД.
        """
        selection = Selection.objects.create(
            product=self.product,
            task_type=SelectionTaskType.COMPOSITION_ANALYSIS,
        )

        # изначально статус QUEUE
        self.assertEqual(selection.selection_status, SelectionStatus.QUEUE)

        # вызываем метод
        selection.mark_as_processing()

        # обновляем объект из БД
        selection.refresh_from_db()

        # проверяем новый статус
        self.assertEqual(selection.selection_status, SelectionStatus.PROCESS)
