from dataclasses import dataclass
from decimal import Decimal
from typing import List, Optional


@dataclass(frozen=True, slots=True)
class ProductData:
    """
    DTO (Data Transfer Object) для передачи данных о продукте
    вне контекста ORM-модели Django.

    Используется для анализа, формирования подборок и других
    операций, где нужна только информация о продукте без привязки
    к инфраструктуре (БД, ORM).
    """
    name: str  # Название продукта
    article_ga: str #Артикул на продукт в ЗЯ
    image_key: Optional[str]  # Ключ изображения на S3
    product_type: Optional[str] # Тип продукта сокращенно
    product_type_detailed: Optional[str]  # Детализированный тип продукта
    measure_value: Optional[Decimal]  # Количество меры (объём, количество штук и т.д.)
    measure_unit: Optional[str]  # Единица измерения (мл, шт и т.д.)
    price_rub: Optional[Decimal]  # Цена в рублях
    ingredients_list: Optional[List[str]]  # Список ингредиентов в виде строк

# apps/products/selectors/product.py
from apps.products.models import Product


def get_product_by_link_ga(
        link: str,
) -> Product | None:
    """
    Возвращает продукт по ссылке из Золотого Яблока,
    если оно существует в базе.

    :param link: Ссылка на средство
    :return: Экземпляр Product или None, если не найден
    """
    try:
        return Product.objects.get(link_ga=link)
    except Product.DoesNotExist:
        return None


def get_product_by_id(
        product_id: int,
) -> Product | None:
    """
    Возвращает продукт по его ID.

    :param product_id: ID продукта
    :return: Экземпляр Product или None, если не найден
    """
    try:
        return Product.objects.get(id=product_id)
    except Product.DoesNotExist:
        return None

# apps/products/selectors/tests/test_product.py

from apps.products.models import Product
from apps.products.selectors.product import (
    get_product_by_id,
    get_product_by_link_ga,
)
from shared.base_classes.conftest import (
    MyTestCase,
)


class ProductSelectorTestCase(MyTestCase):
    """
    Класс для тестирования функции get_product_by_link_ga
    """

    def setUp(self):
        self.product = Product.objects.create(
            link_ga="https://goldapple.ru/test-product",
            product_type="Уход",
            product_type_detailed="Крем для лица",
            name="Тестовый крем",
            article_ga="GA98765",
            measure_type="объём",
            measure_value=30,
            measure_unit="мл",
            price_rub=990.00,
            brand="TestBrandSelector",
        )

    def test_get_product_by_link_ga_found(self):
        """
        Проверка, что средство успешно находится по ссылке.
        """
        result = get_product_by_link_ga("https://goldapple.ru/test-product")
        self.assertIsNotNone(result)
        self.assertEqual(result.link_ga, self.product.link_ga)

    def test_get_product_by_link_ga_not_found(self):
        """
        Проверка, что при отсутствии средства возвращается None.
        """
        result = get_product_by_link_ga("https://goldapple.ru/not-found")
        self.assertIsNone(result)

    def test_get_product_by_id_found(self):
        """
        Проверка, что средство успешно находится по ID.
        """
        result = get_product_by_id(self.product.id)
        self.assertIsNotNone(result)
        self.assertEqual(result.id, self.product.id)

    def test_get_product_by_id_not_found(self):
        """
        Проверка, что при отсутствии средства по ID возвращается None.
        """
        result = get_product_by_id(product_id=999999)
        self.assertIsNone(result)

class SelectionModelTestCase(MyTestCase):
    """
    Тесты для модели Selection
    """

    def setUp(self):
        self.user = User.objects.create_user(
            telegram_id=111222333,
            username='user_selection_test',
        )

        self.product = Product.objects.create(
            link_ga="https://goldapple.ru/test-link",
            product_type="Уход",
            product_type_detailed="Крем для лица",
            name="Крем Тестовый",
            article_ga="GA12345",
            measure_type="объём",
            measure_value=50,
            measure_unit="мл",
            price_rub=1290.00,
            brand="TestBrand",
        )

    def test_is_filled_true(self):
        """
        Статус продукта заполнения = DONE — is_filled должен вернуть True.
        """
        self.product.fill_status = ProductFillStatus.DONE
        self.product.save()
        self.assertTrue(self.product.is_filled)

    def test_is_filled_false(self):
        """
        Статус продукта заполнения = EMPTY — is_filled должен вернуть False.
        """
        self.product.fill_status = ProductFillStatus.EMPTY
        self.product.save()
        self.assertFalse(self.product.is_filled)


class ProductGetDataAboutProductTestCase(MyTestCase):
    """
    Тесты для Product.get_data_about_product,
    который возвращает DTO Product.
    """

    def test_get_data_about_product_full(self):
        """
        Проверяет, что метод возвращает корректно заполненный DTO,
        если у продукта заданы все поля.
        """
        product = Product.objects.create(
            link_ga="https://goldapple.ru/p/123",
            product_type="haircare",
            product_type_detailed="shampoo",
            name="Super Shampoo",
            article_ga="GA-123",
            measure_type="объём",
            measure_value=Decimal("250.00"),
            measure_unit="мл",
            price_rub=Decimal("399.00"),
            brand="Acme",
            image_key="media/8b57cf6a-9f46-4fc5-b765-4d279c30eacd.jpg",
            ingredients_list=["Aqua", "SLES"],
        )

        dto = product.get_data_about_product()

        self.assertIsInstance(dto, ProductData)
        self.assertEqual(dto.name, "Super Shampoo")
        self.assertEqual(dto.image_key, "media/8b57cf6a-9f46-4fc5-b765-4d279c30eacd.jpg")
        self.assertEqual(dto.product_type_detailed, "shampoo")
        self.assertEqual(dto.measure_value, Decimal("250.00"))  # Decimal сохраняется
        self.assertEqual(dto.measure_unit, "мл")
        self.assertEqual(dto.price_rub, Decimal("399.00"))
        self.assertEqual(dto.ingredients_list, ["Aqua", "SLES"])
